## 🧠 VIDEO PROCESSOR

### 📁 `video_engine.py`

```python
import cv2
from moviepy.editor import VideoFileClip, ImageSequenceClip
from ai_edit_engine import face_swap_single

def video_face_swap(video_path, source_face_img):
    cap = cv2.VideoCapture(video_path)

    frames = []
    fps = cap.get(cv2.CAP_PROP_FPS)

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        swapped = face_swap_single(source_face_img, frame)
        frames.append(cv2.cvtColor(swapped, cv2.COLOR_BGR2RGB))

    cap.release()

    clip = ImageSequenceClip(frames, fps=fps)
    output_path = "output.mp4"
    clip.write_videofile(output_path)

    return output_path
```
